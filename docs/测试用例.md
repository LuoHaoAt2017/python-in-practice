# 一、Python 自带测试库：unittest
Python 内置，**不用安装**，直接导入就能写测试。

固定结构：
```python
import unittest

class TestYourCode(unittest.TestCase):
    def test_功能1(self):
        # 测试逻辑
        self.assertEqual(实际结果, 期望结果)

if __name__ == '__main__':
    unittest.main()
```

# 二、直接拿你的**学生管理系统**写测试
我直接给你写好**可运行的完整测试代码**，复制就能跑。

## 完整代码（业务代码 + 测试代码）
```python
# students.py  你的业务代码
students = [{"name": "张三", "score": 90}, {"name": "李四", "score": 85}]

def add_stu(name, score):
    for stu in students:
        if stu["name"] == name:
            return False
    students.append({"name": name, "score": score})
    return True

def del_stu(name):
    for stu in students:
        if stu["name"] == name:
            students.remove(stu)
            return True
    return False

def edit_stu(name, score):
    for stu in students:
        if stu["name"] == name:
            stu["score"] = score
            return True
    return False

def get_stu(name):
    for stu in students:
        if stu["name"] == name:
            return stu
    return None

# ---------------------------------------------------
# test_students.py  测试用例
# ---------------------------------------------------
import unittest
import students

class TestStudentSystem(unittest.TestCase):

    # 每个 test_ 开头的方法就是一个测试用例
    def test_add_student(self):
        # 测试添加学生
        result = students.add_stu("王五", 100)
        self.assertEqual(result, True)  # 期望添加成功

    def test_add_duplicate_student(self):
        # 测试重复添加（应该失败）
        result = students.add_stu("张三", 60)
        self.assertEqual(result, False)

    def test_get_student(self):
        stu = students.get_stu("李四")
        self.assertIsNotNone(stu)  # 期望能找到

    def test_get_not_exist(self):
        stu = students.get_stu("不存在的人")
        self.assertIsNone(stu)     # 期望找不到

    def test_edit_score(self):
        students.edit_stu("张三", 99)
        stu = students.get_stu("张三")
        self.assertEqual(stu["score"], 99)

    def test_delete_student(self):
        result = students.del_stu("李四")
        self.assertEqual(result, True)

if __name__ == '__main__':
    unittest.main()
```

# 四、最常用的 4 个断言（记住就够）
```python
# 等于
self.assertEqual(a, b)

# 不等于
self.assertNotEqual(a, b)

# 是 None
self.assertIsNone(x)

# 不是 None
self.assertIsNotNone(x)

# 为 True
self.assertTrue(x)
```

# 六、极简总结
1. Python 自带 `unittest`，**不用安装**
2. 建一个类继承 `unittest.TestCase`
3. 方法以 `test_` 开头就是测试用例
4. 用 `self.assertEqual(实际, 期望)` 判断对错
5. 运行 `unittest.main()` 自动执行所有用例
